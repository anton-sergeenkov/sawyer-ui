# Usage

### Public

```ts
import { UiAlert } from "@/sawyer-ui/ui-kit"
```

### Internal

```ts
import UiAlert from "@/sawyer-ui/ui-kit/basic/UiAlert"
```

### Usage

- Alert может показываться при успешном / неуспешном выполнении запроса на сервер

```tsx
import { useState } from "react"
import { UiAlert } from "@/sawyer-ui/ui-kit"

const SectionStarter: React.FC = () => {
  const [isShow, setIsShow] = useState<boolean>(false)

  return (
    <>
      <UiAlert type="info" variant="standard" isShow>
        Alert, который нельзя закрыть
      </UiAlert>

      <UiAlert
        type="success"
        variant="standard"
        isShow={isShow}
        setIsShow={setIsShow}
      >
        Alert с useState()
      </UiAlert>

      <button onClick={() => setIsShow(true)}>Show</button>
      <button onClick={() => setIsShow(false)}>Hide</button>
    </>
  )
}
```
